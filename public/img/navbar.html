        <div className="flex justify-between items-center">

          </Link>

            href="/cart"
            className="flex items-center space-x-1"

              className="h-6 w-6"
              fill="none"
              viewBox="0 0 24 24"
              stroke="currentColor"
            >
              <path
                strokeLinecap="round"

                strokeWidth={2}

              />
            </svg>
            {itemCount > 0 && (
              <span className="bg-black text-white text-sm rounded-full h-5 w-5 flex items-center justify-center">
                {itemCount}
              </span>
            )}
          </Link>
        </div>
      </div>
    </nav>
  );
}
